def dump()
  result = Hash.new
  result['type'] = 'master'
  result['entries'] = Array.new
  @entries.each do |entry|
    result['entries'] << { 'owner' => entry.owner,
                           'proto' => entry.proto,
                           'type'  => entry.type,
                           'data'  => entry.data }
  end
  return result
end
